Push Content up by changing order when stacked with a media query.

<header>
  <div>Logo</div>
  <div>Links</div>
</header>
<main>
  <div>Aside</div>
  <div class="content">Content</div>
  <div>Aside</div>
</main>
<footer>
  <div>Footer 1</div>
  <div>Footer 2</div>
  <div>Footer 3</div>
  <div>Footer 4</div>
</footer>
header, main, footer {
  display: flex;
  flex-direction: column;
}

.content { order: -1; }

@media all and ( min-width: 600px ) {
  header { flex-direction: row; }
}

@media all and ( min-width: 750px ) {
  main { flex-direction: row; }
  .content {order: 0;}
}

@media all and ( min-width: 900px ) {
  footer { flex-direction: row; }
}
